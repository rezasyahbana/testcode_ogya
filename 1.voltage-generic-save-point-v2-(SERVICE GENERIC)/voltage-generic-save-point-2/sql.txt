create table:

CREATE TABLE generic_1.voltage_transform_detail_config (
	json_transform_id VARCHAR,
	json_path_field_name VARCHAR,
	transform_type VARCHAR,
	fpe_id VARCHAR,
	CONSTRAINT fk_voltage_transform_detail_config_fpe_id FOREIGN KEY (fpe_id)
        REFERENCES generic_1.voltage_custom_fpe_config(fpe_id)
        ON DELETE CASCADE
);

CREATE TABLE generic_1.voltage_custom_fpe_config (
	fpe_id VARCHAR PRIMARY KEY,
	library_context_id VARCHAR,
	identity VARCHAR,
	shared_secret VARCHAR,
	format VARCHAR,
	CONSTRAINT fk_custom_fpe_library_context_id FOREIGN KEY (library_context_id)
        REFERENCES generic_1.voltage_custom_library_context_config(library_context_id)
        ON DELETE CASCADE
);

CREATE TABLE generic_1.voltage_custom_library_context_config (
	library_context_id VARCHAR PRIMARY KEY,
	policy_url VARCHAR,
	trust_store_path VARCHAR,
	cache_path VARCHAR
);

CREATE TABLE generic_1.SORTING_CONFIG (
	sorting_id VARCHAR PRIMARY KEY,
	is_array VARCHAR,
	is_sort VARCHAR,
	ascdesc VARCHAR,
	sort_by_fieldName VARCHAR
);
